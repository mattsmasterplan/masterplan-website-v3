Getting Started with TypeScript

Basic Types

          
            // Boolean is a datatype where the value can only be true or false
            let isDone: boolean = false;
          
        
          
            // All numbers are floating point values
            // TypeScript supports decimal, hexadecimal, binary, and octal literals
            let decimal: number = 6;
            let hex: number = 0xf00d;
            let binary: number = 0b1010;
            let octal: number = 0o744;
          
        
          
            // String refers to the textual datatype
            // Either double quotes or single quotes work to denote a string
            let color: string = "red";
            let word: string = 'singleQuotes';

            // Template strings can also be declared by enclosing them with the backtick character
            // Strings of this type can contain embedded expressions which will evaluate to the value of the expression
            let fullName: string = 'Bob Bobbington';
            let age: number = 37;
            let templateString: string = `Hello, my name is ${fullName}.
                                         I will be ${age + 1} years old next month.`;
          
        
          
            // There are two ways to declare an array of values
            // The first way is to use the type of the elements followed by []
            let list: number[] = [1, 2, 3];

            // The second way is to use a generic array type
            let list: Array = [1, 2, 3];
          
        
          
            // Tuple datatypes allow you to create an array with a fixed number of elements
            // whose types are known, but not necessarily the same
            let x: [string, number];
            x = ["hello", 10];

            // You can make type specific commands to the known elements
            console.log(x[0].substring(1));
          
        
          
            // The enum datatype is a way to name the elements of a defined set 
            // By default, enums begin numbering their elements starting at 0 
            enum Color {
              Red,
              Green,
              Blue
            }

            // Default numbering can be changed manually 
            enum Color {
              Red = 1,
              Green,
              Blue
            }

            // Access an element like this 
            let c: Color = Color.Green;
            // Or like this 
            let colorName: string = Color[2];
          
        
            
              // Declaring a variable as datatype any opts-out of type checking and lets any value compile
              // This his helpful to describe the type of variables that we do not know 
              let notSure: any = 4;
            
          
            
              // Void is the absence of having any type at all
              // Such as a function that does not return a value 
              function warnUser(): void { console.log("useless function"); }
            
          
          
          // Undefined and Null are each seperate datatypes
          // They are subtypes of all other types. You can assign them to a number for example

          // Not much else we can assign to these variables!
          let u: undefined = undefined;
          let n: null = null;
        
      
            
              // The never type represents values that never occur
              // For instance, never is the return type for a function expression
              // or an arrow function expression that always throws an exception or one that never returns;

              // Function returning never must have unreachable end point
              function error(message: string): never {
              throw new Error(message);
              }
